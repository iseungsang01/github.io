    "import time\n",
    "import threading\n",
    "\n",
    "class Timer:\n",
    "    def __init__(self, interval):\n",
    "        self.interval = interval  # 주기 (초 단위)\n",
    "        self.start_time = time.monotonic()\n",
    "        self.target_time = self.start_time + interval\n",
    "\n",
    "    def is_time_elapsed(self):\n",
    "        current_time = time.monotonic()\n",
    "        elapsed = current_time >= self.target_time\n",
    "        \n",
    "        if elapsed:\n",
    "            # 오차 보정\n",
    "            error = current_time - self.target_time\n",
    "            self.target_time += self.interval - error\n",
    "            return True\n",
    "        return False\n",
    "\n",
    "def run_timer(name, interval):\n",
    "    timer = Timer(interval)\n",
    "    while True:\n",
    "        if timer.is_time_elapsed():\n",
    "            print(f\"{name} - Timer Event at {time.strftime('%H:%M:%S.%f')}\")\n",
    "\n",
    "# 첫 번째 작업: 1초에 30회 실행 (0.0333초 간격)\n",
    "thread_1 = threading.Thread(target=run_timer, args=(\"Task 1 (30 times/sec)\", 1 / 30))\n",
    "\n",
    "# 두 번째 작업: 96/BPM 초에 한 번 실행 (BPM이 120이라면 96/120 = 0.8초 간격)\n",
    "BPM = 120  # 예시로 120 BPM 설정\n",
    "interval_2 = 96 / BPM\n",
    "thread_2 = threading.Thread(target=run_timer, args=(\"Task 2 (96/BPM interval)\", interval_2))\n",
    "\n",
    "# 스레드 시작\n",
    "thread_1.start()\n",
    "thread_2.start()\n",
    "\n",
    "# 메인 스레드에서 스레드가 완료될 때까지 대기\n",
    "thread_1.join()\n",
    "thread_2.join()\n"
